@app.route('/kyc', methods=['POST'])
def upload_kyc_file():
    app.logger.info('KYC file upload API accessed.')
    if 'file' not in request.files:
        return jsonify({'error': 'No file part'}), 400

    file = request.files['file']
    if file.filename == '':
        return jsonify({'error': 'No selected file'}), 400

    filename = secure_filename(file.filename)
    file_path = os.path.join(app.config['UPLOAD_FOLDER'], filename)
    file.save(file_path)
    app.logger.info(f'KYC file saved: {filename}')

    try:
        extracted_data = process_file(file_path)
        if not extracted_data:
            return jsonify({'error': 'Failed to extract data from file.'}), 500

        session['extracted_data'] = extracted_data
        return jsonify({'extracted_data': extracted_data}), 200
    except Exception as e:
        app.logger.error(f"Error processing KYC file: {e}")
        return jsonify({'error': str(e)}), 500

@app.route('/cv', methods=['POST'])
def cv_data_extraction():
    if "pdf_file" not in request.files:
        return jsonify({"error": "No file uploaded"}), 400

    uploaded_file = request.files["pdf_file"]
    if uploaded_file.filename == "":
        return jsonify({"error": "No file selected"}), 400

    filename = secure_filename(uploaded_file.filename)
    ext = os .path.splitext(filename)[1].lower()

    unique_id = str(uuid.uuid4())[:8]
    base_filename = f"{os.path.splitext(filename)[0]}_{unique_id}"
    file_path = os.path.join(UPLOAD_FOLDER, filename)

    # Delete the file if it already exists
    if os.path.exists(file_path):
        try:
            os.remove(file_path)
        except PermissionError:
            return jsonify({"error": "File is locked or in use. Please try again after closing it."}), 500

    # Save the uploaded file
    try:
        uploaded_file.save(file_path)
    except PermissionError:
        return jsonify({"error": "Permission denied when saving file. Please ensure it is not open."}), 500

    # Convert DOCX to PDF if needed
    if ext == ".docx":
        pdf_path = os.path.join(UPLOAD_FOLDER, base_filename + ".pdf")
        try:
            safe_docx_to_pdf(file_path, pdf_path)
            os.remove(file_path)  # Optionally delete the original .docx
        except Exception as e:
            return jsonify({"error": f"Failed to convert Word to PDF: {str(e)}"}), 500
    elif ext == ".pdf":
        pdf_path = file_path
    else:
        return jsonify({"error": "Unsupported file format. Only .pdf or .docx allowed."}), 400

    try:
        # Convert PDF to images (pages)
        image_paths = pdf_to_image_cv(pdf_path)
        combined_data = {
            "name": None,
            "phoneNumber": None,
            "email": None,
            "location": None,
            "maritalStatus": None,
            "presentEmployer": None,
            "presentJobTitle": None,
            "presentJobDescription": None,
            "presentJobStartingDate": None,
            "totalExperience": None,
            "dateOfBirth": None,
            "skills": [],
            "pastWorkExperience": [],
            "educationalQualification": [],
            "yearGap":None
        }  # Initialize with empty or None values

        # Process each image (page)
        for image_path in image_paths:
            output_json = gemini_output_cv(image_path, cv_system_prompt, cv_user_prompt)

            try:
                json_data = json.loads(output_json)

                # Combine data only if not already present
                if json_data.get("name") and not combined_data["name"]:
                    combined_data["name"] = json_data["name"]
                if json_data.get("phoneNumber") and not combined_data["phoneNumber"]:
                    combined_data["phoneNumber"] = json_data["phoneNumber"]
                if json_data.get("email") and not combined_data["email"]:
                    combined_data["email"] = json_data["email"]
                if json_data.get("location") and not combined_data["location"]:
                    combined_data["location"] = json_data["location"]
                if json_data.get("maritalStatus") and not combined_data["maritalStatus"]:
                    combined_data["maritalStatus"] = json_data["maritalStatus"]
                if json_data.get("presentEmployer") and not combined_data["presentEmployer"]:
                    combined_data["presentEmployer"] = json_data["presentEmployer"]
                if json_data.get("presentJobTitle") and not combined_data["presentJobTitle"]:
                    combined_data["presentJobTitle"] = json_data["presentJobTitle"]
                if json_data.get("presentJobDescription") and not combined_data["presentJobDescription"]:
                    combined_data["presentJobDescription"] = json_data["presentJobDescription"]
                if json_data.get("presentJobStartingDate") and not combined_data["presentJobStartingDate"]:
                    combined_data["presentJobStartingDate"] = json_data["presentJobStartingDate"]
                if json_data.get("totalExperience") and not combined_data["totalExperience"]:
                    combined_data["totalExperience"] = json_data["totalExperience"]
                if json_data.get("dateOfBirth") and not combined_data["dateOfBirth"]:
                    combined_data["dateOfBirth"] = json_data["dateOfBirth"]
                if json_data.get("yearGap") and not combined_data["yearGap"]:
                    combined_data["yearGap"] = json_data["yearGap"]

                # Combine skills (ensure no duplicates)
                if json_data.get("skills"):
                    for skill in json_data["skills"]:
                        if skill not in combined_data["skills"]:
                            combined_data["skills"].append(skill)

                # Combine past work experience
                if json_data.get("Past Work Experience 1"):
                    for i in range(1, 5):
                        if json_data.get(f"Past Work Experience {i}"):
                            past_work_exp = json_data[f"Past Work Experience {i}"]
                            formatted_exp = {
                                "company": past_work_exp.get("company"),
                                "description": past_work_exp.get("description"),
                                "duration": past_work_exp.get("duration"),
                                "jobTitle": past_work_exp.get("jobTitle")
                            }
                            # Avoid duplicates in pastWorkExperience
                            if formatted_exp not in combined_data["pastWorkExperience"]:
                                combined_data["pastWorkExperience"].append(formatted_exp)

                # Combine educational qualifications (ensure no duplicates)
                if json_data.get("educationalQualification"):
                    for qualification in json_data["educationalQualification"]:
                        if qualification not in combined_data["educationalQualification"]:
                            formatted_edu = {
                                "details": qualification.get("details"),
                                "institution": qualification.get("institution"),
                                "level": qualification.get("level"),
                                "year": qualification.get("year")
                            }
                            combined_data["educationalQualification"].append(formatted_edu)
            # print(combined_data)
            except json.JSONDecodeError as e:
                print(f"Error parsing JSON for {image_path}: {e}")
        # a = {"extracted_data": [combined_data]}
        # call_cv_candidate_function(json.dumps(a))
        # Return the combined data
        print({"extracted_data": [combined_data]})
        return jsonify({"extracted_data": [combined_data]}), 200

    except Exception as e:
        return jsonify({"error": str(e)}), 500
